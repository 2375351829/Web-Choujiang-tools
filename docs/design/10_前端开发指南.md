### 1.8 src/composables/useSocket.js
```javascript
import { ref, onMounted, onUnmounted } from 'vue'
import { io } from 'socket.io-client'

export function useSocket() {
  const socket = ref(null)
  const isConnected = ref(false)

  onMounted(() => {
    socket.value = io()

    socket.value.on('connect', () => {
      isConnected.value = true
      console.log('Socket.io 已连接')
    })

    socket.value.on('disconnect', () => {
      isConnected.value = false
      console.log('Socket.io 已断开')
    })

    socket.value.on('draw:result', (data) => {
      console.log('抽奖结果:', data)
    })
  })

  onUnmounted(() => {
    if (socket.value) socket.value.disconnect()
  })

  const emit = (event, data) => {
    if (socket.value && isConnected.value) {
      socket.value.emit(event, data)
    }
  }

  return { socket, isConnected, emit }
}
```

            </el-descriptions-item>

              <el-tag :type="drawStatusType">{{ drawStatusText }}</el-tag>
            </el-descriptions-item>
          </el-descriptions>
        </el-card>

        <el-card header="最近中奖" style="margin-top: 20px">
          <el-table :data="recentWinners" stripe size="small" max-height="300">
            <el-table-column prop="participant_name" label="中奖者" />

import { ref, computed, onMounted } from 'vue'
import { getSessions, getSession, drawLottery } from '../api/sessions.js'
import { useSocket } from '../composables/useSocket.js'

const { isConnected, emit } = useSocket()

const sessions = ref([])
const prizes = ref([])
const recentWinners = ref([])
const isDrawing = ref(false)
const drawStatus = ref('idle') // idle | drawing | paused

const form = ref({ session_id: null, prize_id: null })

const drawStatusText = computed(() => {
  const map = { idle: '已停止', drawing: '抽奖中', paused: '已暂停' }
  return map[drawStatus.value] || '已停止'
})
const drawStatusType = computed(() => {
  const map = { idle: 'info', drawing: 'success', paused: 'warning' }
  return map[drawStatus.value] || 'info'
})

const loadSessions = async () => {
  sessions.value = await getSessions()
}

const loadSessionDetail = async () => {
  if (!form.value.session_id) return
  const detail = await getSession(form.value.session_id)
  prizes.value = detail.prizes || []
  recentWinners.value = detail.results || []
}

const startDraw = async () => {
  if (!form.value.session_id || !form.value.prize_id) return
  isDrawing.value = true
  drawStatus.value = 'drawing'
  emit('draw:start', { session_id: form.value.session_id, prize_id: form.value.prize_id })
}

const pauseDraw = () => {
  drawStatus.value = 'paused'
  emit('draw:pause')
}

const stopDraw = async () => {
  const result = await drawLottery(form.value.session_id, form.value.prize_id)
  isDrawing.value = false
  drawStatus.value = 'idle'
  await loadSessionDetail()
}

const resetDraw = () => {
  isDrawing.value = false
  drawStatus.value = 'idle'
  emit('draw:reset')
}

onMounted(() => {
  loadSessions()
})
</script>

<template>
  <div class="lottery-display" :style="bgStyle">
    <!-- 连接状态 -->
    <div class="connection-status" :class="{ connected: isConnected }">
      {{ isConnected ? '已连接' : '未连接' }}
    </div>

    <!-- 参与者轮播 -->

      <!-- Grid动画 -->
      <div v-if="config.animation_type === 'grid'" class="grid-animation">
        <div v-for="(name, index) in gridNames" :key="index"
          class="grid-cell" :class="{ active: isDrawing && activeIndex === index, winner: !isDrawing && winnerIndex === index }">
          {{ name }}
        </div>
      </div>

      <!-- List动画 -->
      <div v-if="config.animation_type === 'list'" class="list-animation">
        <div v-for="(name, index) in displayNames" :key="index"
          class="list-item" :class="{ active: isDrawing && activeIndex === index, winner: !isDrawing && winnerIndex === index }">
          {{ name }}
        </div>
      </div>
    </div>

    <!-- 中奖结果展示 -->
    <transition name="fade">
      <div v-if="currentWinner" class="winner-display">
        <div class="winner-name">{{ currentWinner.name }}</div>
        <div class="winner-prize">{{ currentWinner.prizeName }}</div>
      </div>
    </transition>

    <!-- 中奖名单 -->
    <div v-if="config.show_winner && winners.length > 0" class="winner-list">
      <span v-for="(w, i) in winners" :key="i" class="winner-tag">
        {{ w.name }} ({{ w.prizeName }})
      </span>

<script setup>
import { ref, computed, onMounted, onUnmounted } from 'vue'
import { io } from 'socket.io-client'
import confetti from 'canvas-confetti'

const socket = ref(null)
const isConnected = ref(false)
const isDrawing = ref(false)
const activeIndex = ref(0)
const winnerIndex = ref(-1)
const currentWinner = ref(null)
const participants = ref([])
const winners = ref([])
const config = ref({
  title: '抽奖活动',
  background_type: 'gradient',
  background_color: '#0f172a',
  gradient_start: '#667eea',
  gradient_end: '#764ba2',
  gradient_degree: 135,
  animation_type: 'grid',
  show_participants: true,
  show_prizes: true,
  show_winner: true,
  carousel_speed: 3000
})

let animationTimer = null

const displayNames = computed(() => participants.value.map(p => p.name))
const gridNames = computed(() => {
  const names = displayNames.value
  // Grid模式取前9个
  return names.slice(0, 9)
})

const bgStyle = computed(() => {
  const c = config.value
  if (c.background_type === 'gradient') {
    return { background: `linear-gradient(${c.gradient_degree}deg, ${c.gradient_start}, ${c.gradient_end})` }
  }
  if (c.background_type === 'color') {
    return { background: c.background_color }
  }
  return { background: '#0f172a' }
})

const startAnimation = () => {
  const total = config.value.animation_type === 'grid' ? gridNames.value.length : displayNames.value.length
  if (total === 0) return
  animationTimer = setInterval(() => {
    activeIndex.value = (activeIndex.value + 1) % total
  }, 80)
}

const stopAnimation = (winnerName) => {
  clearInterval(animationTimer)
  animationTimer = null
  isDrawing.value = false

  // 找到中奖者索引
  const idx = displayNames.value.indexOf(winnerName)
  if (idx >= 0) winnerIndex.value = idx

  // 彩带效果
  confetti({ particleCount: 200, spread: 90, origin: { y: 0.6 } })
}

const resetAnimation = () => {
  clearInterval(animationTimer)
  animationTimer = null
  isDrawing.value = false
  activeIndex.value = 0
  winnerIndex.value = -1
  currentWinner.value = null
}

onMounted(async () => {
  // 加载配置
  try {
    const res = await fetch('/api/config')
    if (res.ok) config.value = await res.json()
  } catch (e) {}

  // 加载参与者
  if (config.value.current_session_id) {
    try {
      const res = await fetch(`/api/sessions/${config.value.current_session_id}`)
      if (res.ok) {
        const data = await res.json()
        participants.value = data.participants || []
      }
    } catch (e) {}
  }

  // Socket.io连接
  socket.value = io()
  socket.value.on('connect', () => { isConnected.value = true })
  socket.value.on('disconnect', () => { isConnected.value = false })

  socket.value.on('draw:start', () => {
    isDrawing.value = true
    winnerIndex.value = -1
    currentWinner.value = null
    startAnimation()
  })

  socket.value.on('draw:pause', () => {
    clearInterval(animationTimer)
  })

  socket.value.on('draw:result', (data) => {
    const winnerName = data.winner?.name
    stopAnimation(winnerName)
    currentWinner.value = { name: winnerName, prizeName: data.prize?.name }
    winners.value.unshift({ name: winnerName, prizeName: data.prize?.name })
  })

  socket.value.on('draw:reset', () => {
    resetAnimation()
  })

  socket.value.on('config:update', (data) => {
    config.value = data.config
  })
})

onUnmounted(() => {
  if (animationTimer) clearInterval(animationTimer)
  if (socket.value) socket.value.disconnect()
})
</script>
