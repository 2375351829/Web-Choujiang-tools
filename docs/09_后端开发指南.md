### 2.1 src/db/init.js（完整代码）

**文件路径**: `server/src/db/init.js`

```javascript
import Database from 'better-sqlite3';
import path from 'path';
import { fileURLToPath } from 'url';

const __dirname = path.dirname(fileURLToPath(import.meta.url));
const DB_PATH = process.env.DB_PATH || path.join(__dirname, '../../data/lottery.db');

let db;

export function getDb() {
  if (!db) {
    db = new Database(DB_PATH);
    db.pragma('journal_mode = WAL');
    db.pragma('foreign_keys = ON');
  }
  return db;
}

export function initDatabase() {
  const db = getDb();

  db.exec(`
    CREATE TABLE IF NOT EXISTS participants (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      name TEXT NOT NULL,
      email TEXT DEFAULT '',
      phone TEXT DEFAULT '',
      avatar TEXT DEFAULT '',
      status TEXT DEFAULT '未中奖' CHECK(status IN ('未中奖', '已中奖')),
      is_blacklisted INTEGER DEFAULT 0 CHECK(is_blacklisted IN (0, 1)),
      weight INTEGER DEFAULT 1 CHECK(weight >= 0),
      created_at DATETIME DEFAULT CURRENT_TIMESTAMP
    );

    CREATE TABLE IF NOT EXISTS prizes (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      name TEXT NOT NULL,
      description TEXT DEFAULT '',
      image TEXT DEFAULT '',
      weight INTEGER DEFAULT 1,
      draw_count INTEGER DEFAULT 1 CHECK(draw_count > 0),
      created_at DATETIME DEFAULT CURRENT_TIMESTAMP
    );

    CREATE TABLE IF NOT EXISTS lottery_sessions (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      name TEXT NOT NULL,
      description TEXT DEFAULT '',
      status TEXT DEFAULT '未开始' CHECK(status IN ('未开始', '进行中', '已结束')),
      created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
      updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
    );

    CREATE TABLE IF NOT EXISTS session_participants (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      session_id INTEGER NOT NULL,
      participant_id INTEGER NOT NULL,
      FOREIGN KEY (session_id) REFERENCES lottery_sessions(id) ON DELETE CASCADE,
      FOREIGN KEY (participant_id) REFERENCES participants(id) ON DELETE CASCADE,
      UNIQUE(session_id, participant_id)
    );

    CREATE TABLE IF NOT EXISTS session_prizes (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      session_id INTEGER NOT NULL,
      prize_id INTEGER NOT NULL,
      quantity INTEGER DEFAULT 1 CHECK(quantity > 0),
      FOREIGN KEY (session_id) REFERENCES lottery_sessions(id) ON DELETE CASCADE,
      FOREIGN KEY (prize_id) REFERENCES prizes(id) ON DELETE CASCADE,
      UNIQUE(session_id, prize_id)
    );

    CREATE TABLE IF NOT EXISTS session_results (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      session_id INTEGER NOT NULL,
      participant_id INTEGER NOT NULL,
      prize_id INTEGER NOT NULL,
      drawn_at DATETIME DEFAULT CURRENT_TIMESTAMP,
      FOREIGN KEY (session_id) REFERENCES lottery_sessions(id) ON DELETE CASCADE,
      FOREIGN KEY (participant_id) REFERENCES participants(id) ON DELETE CASCADE,
      FOREIGN KEY (prize_id) REFERENCES prizes(id) ON DELETE CASCADE,
      UNIQUE(session_id, participant_id, prize_id)
    );

    CREATE TABLE IF NOT EXISTS lottery_config (
      id INTEGER PRIMARY KEY DEFAULT 1,
      background_type TEXT DEFAULT 'gradient' CHECK(background_type IN ('gradient', 'color', 'image')),
      background_image TEXT DEFAULT '',
      background_color TEXT DEFAULT '#0f172a',
      gradient_start TEXT DEFAULT '#667eea',
      gradient_end TEXT DEFAULT '#764ba2',
      gradient_degree INTEGER DEFAULT 135 CHECK(gradient_degree BETWEEN 0 AND 360),
      show_participants INTEGER DEFAULT 1,
      show_prizes INTEGER DEFAULT 1,
      show_winner INTEGER DEFAULT 1,
      carousel_speed INTEGER DEFAULT 3000 CHECK(carousel_speed BETWEEN 500 AND 10000),
      title TEXT DEFAULT '抽奖活动',
      animation_type TEXT DEFAULT 'grid' CHECK(animation_type IN ('grid', 'sphere', 'list')),
      current_session_id INTEGER,
      theme TEXT DEFAULT 'dark' CHECK(theme IN ('dark', 'light', 'ocean', 'forest', 'sunset', 'purple'))
    );

    -- 索引
    CREATE INDEX IF NOT EXISTS idx_participants_status ON participants(status);
    CREATE INDEX IF NOT EXISTS idx_participants_blacklisted ON participants(is_blacklisted);
    CREATE INDEX IF NOT EXISTS idx_results_session ON session_results(session_id);
    CREATE INDEX IF NOT EXISTS idx_results_participant ON session_results(participant_id);
    CREATE INDEX IF NOT EXISTS idx_results_prize ON session_results(prize_id);
  `);

  // 触发器：中奖时自动更新参与者状态
  db.exec(`
    CREATE TRIGGER IF NOT EXISTS trg_participant_status_sync
    AFTER INSERT ON session_results
    FOR EACH ROW
    BEGIN
      UPDATE participants SET status = '已中奖'
      WHERE id = NEW.participant_id;
    END;

    CREATE TRIGGER IF NOT EXISTS trg_participant_result_delete
    AFTER DELETE ON session_results
    FOR EACH ROW
    BEGIN
      UPDATE participants SET status = '未中奖'
      WHERE id = OLD.participant_id
      AND id NOT IN (SELECT participant_id FROM session_results);
    END;

    CREATE TRIGGER IF NOT EXISTS trg_session_updated_at
    AFTER UPDATE ON lottery_sessions
    FOR EACH ROW
    BEGIN
      UPDATE lottery_sessions SET updated_at = CURRENT_TIMESTAMP
      WHERE id = NEW.id;
    END;
  `);

  // 插入默认配置
  db.prepare('INSERT OR IGNORE INTO lottery_config (id) VALUES (1)').run();

  console.log('数据库初始化完成');
}
```

```javascript
import { Router } from 'express';
import { getDb } from '../db/init.js';
import multer from 'multer';
import XLSX from 'xlsx';

const router = Router();

// GET /api/participants - 获取人员列表
router.get('/', (req, res) => {
  const db = getDb();
  const { status, blacklist, search, page = 1, pageSize = 50 } = req.query;

  let sql = 'SELECT * FROM participants WHERE 1=1';
  const params = [];

  if (status) {
    sql += ' AND status = ?';
    params.push(status);
  }
  if (blacklist !== undefined) {
    sql += ' AND is_blacklisted = ?';
    params.push(Number(blacklist));
  }
  if (search) {
    sql += ' AND (name LIKE ? OR email LIKE ? OR phone LIKE ?)';
    const keyword = `%${search}%`;
    params.push(keyword, keyword, keyword);
  }

  // 分页
  const offset = (Number(page) - 1) * Number(pageSize);
  sql += ' ORDER BY created_at DESC LIMIT ? OFFSET ?';
  params.push(Number(pageSize), offset);

  try {
    const participants = db.prepare(sql).all(...params);
    res.json(participants);
  } catch (err) {
    res.status(500).json({ success: false, error: err.message });
  }
});

// GET /api/participants/:id
router.get('/:id', (req, res) => {
  const db = getDb();
  const participant = db.prepare('SELECT * FROM participants WHERE id = ?').get(req.params.id);
  if (!participant) {
    return res.status(404).json({ success: false, error: '人员不存在' });
  }
  res.json(participant);
});

// POST /api/participants - 创建人员
router.post('/', (req, res) => {
  const db = getDb();
  const { name, email = '', phone = '', weight = 1 } = req.body;

  if (!name || !name.trim()) {
    return res.status(400).json({ success: false, error: '姓名不能为空' });
  }

  // 重复检查
  const existing = db.prepare('SELECT id FROM participants WHERE name = ? AND email = ?').get(name.trim(), email.trim());
  if (existing) {
    return res.status(409).json({ success: false, error: '该姓名和邮箱组合已存在', existing_id: existing.id });
  }

  try {
    const result = db.prepare(
      'INSERT INTO participants (name, email, phone, weight) VALUES (?, ?, ?, ?)'
    ).run(name.trim(), email.trim(), phone.trim(), Number(weight));

    const participant = db.prepare('SELECT * FROM participants WHERE id = ?').get(result.lastInsertRowid);
    res.status(201).json(participant);
  } catch (err) {
    res.status(500).json({ success: false, error: err.message });
  }
});

// PUT /api/participants/:id
router.put('/:id', (req, res) => {
  const db = getDb();
  const { name, email, phone, weight } = req.body;
  const existing = db.prepare('SELECT * FROM participants WHERE id = ?').get(req.params.id);
  if (!existing) {
    return res.status(404).json({ success: false, error: '人员不存在' });
  }

  try {
    db.prepare(
      'UPDATE participants SET name = COALESCE(?, name), email = COALESCE(?, email), phone = COALESCE(?, phone), weight = COALESCE(?, weight) WHERE id = ?'
    ).run(name, email, phone, weight ? Number(weight) : null, req.params.id);

    const participant = db.prepare('SELECT * FROM participants WHERE id = ?').get(req.params.id);
    res.json(participant);
  } catch (err) {
    res.status(500).json({ success: false, error: err.message });
  }
});

// DELETE /api/participants/:id
router.delete('/:id', (req, res) => {
  const db = getDb();
  const existing = db.prepare('SELECT * FROM participants WHERE id = ?').get(req.params.id);
  if (!existing) {
    return res.status(404).json({ success: false, error: '人员不存在' });
  }
  db.prepare('DELETE FROM participants WHERE id = ?').run(req.params.id);
  res.json({ success: true, message: '人员已删除' });
});

// PUT /api/participants/:id/status
router.put('/:id/status', (req, res) => {
  const db = getDb();
  const { status } = req.body;
  if (!['已中奖', '未中奖'].includes(status)) {
    return res.status(400).json({ success: false, error: '状态值非法' });
  }
  db.prepare('UPDATE participants SET status = ? WHERE id = ?').run(status, req.params.id);
  res.json({ success: true, message: '状态已更新' });
});

// PUT /api/participants/:id/blacklist
router.put('/:id/blacklist', (req, res) => {
  const db = getDb();
  const { is_blacklisted } = req.body;
  db.prepare('UPDATE participants SET is_blacklisted = ? WHERE id = ?').run(Number(is_blacklisted), req.params.id);
  res.json({ success: true, message: '黑名单状态已更新' });
});

// POST /api/participants/import - 批量导入
const upload = multer({ dest: 'uploads/', limits: { fileSize: 10 * 1024 * 1024 } });
router.post('/import', upload.single('file'), (req, res) => {
  const db = getDb();
  const { session_id } = req.body;

  try {
    const workbook = XLSX.readFile(req.file.path);
    const sheet = workbook.Sheets[workbook.SheetNames[0]];
    const rows = XLSX.utils.sheet_to_json(sheet);

    let imported = 0;
    let skipped = 0;

    const insertStmt = db.prepare(
      'INSERT OR IGNORE INTO participants (name, email, phone) VALUES (?, ?, ?)'
    );
    const linkStmt = db.prepare(
      'INSERT OR IGNORE INTO session_participants (session_id, participant_id) VALUES (?, ?)'
    );

    const transaction = db.transaction(() => {
      for (const row of rows) {
        const name = (row.name || '').toString().trim();
        if (!name) { skipped++; continue; }

        const email = (row.email || '').toString().trim();
        const phone = (row.phone || '').toString().trim();

        const existing = db.prepare('SELECT id FROM participants WHERE name = ? AND email = ?').get(name, email);
        let participantId;

        if (existing) {
          participantId = existing.id;
          skipped++;
        } else {
          const result = insertStmt.run(name, email, phone);
          participantId = result.lastInsertRowid;
          imported++;
        }

        if (session_id) {
          linkStmt.run(Number(session_id), participantId);
        }
      }
    });
    transaction();

    res.json({ success: true, imported, skipped, message: `导入完成` });
  } catch (err) {
    res.status(500).json({ success: false, error: err.message });
  }
});

export default router;
```

### 3.2 src/routes/sessions.js（完整代码，包含抽奖算法）

**文件路径**: `server/src/routes/sessions.js`

```javascript
import { Router } from 'express';
import { getDb } from '../db/init.js';

const router = Router();

// 加权随机抽奖算法
function weightedDraw(participants) {
  if (participants.length === 0) return null;
  if (participants.length === 1) return participants[0];

  const totalWeight = participants.reduce((sum, p) => sum + (p.weight || 1), 0);
  if (totalWeight <= 0) {
    return participants[Math.floor(Math.random() * participants.length)];
  }

  let random = Math.random() * totalWeight;
  for (const p of participants) {
    random -= (p.weight || 1);
    if (random <= 0) return p;
  }
  return participants[participants.length - 1];
}

// GET /api/sessions
router.get('/', (req, res) => {
  const db = getDb();
  const sessions = db.prepare('SELECT * FROM lottery_sessions ORDER BY created_at DESC').all();
  res.json(sessions);
});

// GET /api/sessions/:id - 获取场次详情
router.get('/:id', (req, res) => {
  const db = getDb();
  const session = db.prepare('SELECT * FROM lottery_sessions WHERE id = ?').get(req.params.id);
  if (!session) return res.status(404).json({ success: false, error: '场次不存在' });

  const participants = db.prepare(`
    SELECT p.* FROM participants p
    JOIN session_participants sp ON p.id = sp.participant_id
    WHERE sp.session_id = ?
  `).all(req.params.id);

  const prizes = db.prepare(`
    SELECT pr.*, sp.quantity, sp.quantity - COUNT(sr.id) as remaining
    FROM prizes pr
    JOIN session_prizes sp ON pr.id = sp.prize_id
    LEFT JOIN session_results sr ON sp.session_id = sr.session_id AND sp.prize_id = sr.prize_id
    WHERE sp.session_id = ?
    GROUP BY sp.prize_id
  `).all(req.params.id);

  const results = db.prepare(`
    SELECT sr.drawn_at, p.name as participant_name, p.email, p.phone, pr.name as prize_name
    FROM session_results sr
    JOIN participants p ON sr.participant_id = p.id
    JOIN prizes pr ON sr.prize_id = pr.id
    WHERE sr.session_id = ?
    ORDER BY sr.drawn_at DESC
  `).all(req.params.id);

  res.json({ ...session, participants, prizes, results });
});

// POST /api/sessions
router.post('/', (req, res) => {
  const db = getDb();
  const { name, description = '' } = req.body;
  if (!name) return res.status(400).json({ success: false, error: '场次名称不能为空' });

  const result = db.prepare('INSERT INTO lottery_sessions (name, description) VALUES (?, ?)').run(name, description);
  const session = db.prepare('SELECT * FROM lottery_sessions WHERE id = ?').get(result.lastInsertRowid);
  res.status(201).json(session);
});

// POST /api/sessions/:id/participants
router.post('/:id/participants', (req, res) => {
  const db = getDb();
  const { participant_ids } = req.body;
  if (!Array.isArray(participant_ids)) return res.status(400).json({ success: false, error: 'participant_ids必须为数组' });

  const insert = db.prepare('INSERT OR IGNORE INTO session_participants (session_id, participant_id) VALUES (?, ?)');
  const addMany = db.transaction((ids) => {
    for (const id of ids) insert.run(req.params.id, id);
  });
  addMany(participant_ids);

  res.json({ success: true, added: participant_ids.length, message: '参与者已添加到场次' });
});

// POST /api/sessions/:id/prizes
```

```javascript
/**
 * 请求参数校验中间件
 * 用法: router.post('/xxx', validate({ body: { name: 'required|string' } }), handler)
 */
export function validate({ body, params, query }) {
  return (req, res, next) => {
    const errors = [];

    if (body) {
      for (const [field, rules] of Object.entries(body)) {
        const ruleList = rules.split('|');
        const value = req.body[field];

        if (ruleList.includes('required') && (value === undefined || value === null || value === '')) {
          errors.push(`${field} 不能为空`);
        }
        if (ruleList.includes('string') && value !== undefined && typeof value !== 'string') {
          errors.push(`${field} 必须是字符串`);
        }
        if (ruleList.includes('number') && value !== undefined && isNaN(Number(value))) {
          errors.push(`${field} 必须是数字`);
        }
        if (ruleList.includes('array') && value !== undefined && !Array.isArray(value)) {
          errors.push(`${field} 必须是数组`);
        }
      }
    }

    if (params) {
      for (const [field, rules] of Object.entries(params)) {
        const ruleList = rules.split('|');
        const value = req.params[field];

        if (ruleList.includes('required') && !value) {
          errors.push(`参数 ${field} 不能为空`);
        }
        if (ruleList.includes('number') && value && isNaN(Number(value))) {
          errors.push(`参数 ${field} 必须是数字`);
        }
      }
    }

    if (errors.length > 0) {
      return res.status(400).json({ success: false, error: errors.join('; ') });
    }

    next();
  };
}
```

```javascript
/**
 * 构建动态 WHERE 子句
 */
export function buildWhereClause(conditions) {
  const clauses = [];
  const params = [];

  for (const { field, operator = '=', value, like = false } of conditions) {
    if (value === undefined || value === null || value === '') continue;

    if (like) {
      clauses.push(`${field} LIKE ?`);
      params.push(`%${value}%`);
    } else {
      clauses.push(`${field} ${operator} ?`);
      params.push(value);
    }
  }

  const where = clauses.length > 0 ? `WHERE ${clauses.join(' AND ')}` : '';
  return { where, params };
}
```

```javascript
/**
 * 抽奖核心服务
 * 将抽奖逻辑从路由中解耦，便于测试和复用
 */

/**
 * 加权随机抽奖算法
 * @param {Array} participants - 参与者数组，每个元素需包含 weight 字段
 * @returns {Object|null} 中奖者对象
 */
export function weightedDraw(participants) {
  if (participants.length === 0) return null;
  if (participants.length === 1) return participants[0];

  const totalWeight = participants.reduce((sum, p) => sum + (p.weight || 1), 0);
  if (totalWeight <= 0) {
    return participants[Math.floor(Math.random() * participants.length)];
  }

  let random = Math.random() * totalWeight;
  for (const p of participants) {
    random -= (p.weight || 1);
    if (random <= 0) return p;
  }
  return participants[participants.length - 1];
}

/**
 * 获取场次中某奖品的可参与抽奖人员
 */
export function getEligibleParticipants(sessionId, prizeId) {
  const db = getDb();
  return db.prepare(`
    SELECT p.* FROM participants p
    JOIN session_participants sp ON p.id = sp.participant_id
    WHERE sp.session_id = ? AND p.is_blacklisted = 0
    AND p.id NOT IN (
      SELECT participant_id FROM session_results
      WHERE session_id = ? AND prize_id = ?
    )
  `).all(sessionId, sessionId, prizeId);
}

/**
 * 获取奖品剩余数量
 */
export function getPrizeRemaining(sessionId, prizeId) {
  const db = getDb();
  const result = db.prepare(`
    SELECT sp.quantity - COUNT(sr.id) as remaining
    FROM session_prizes sp
    LEFT JOIN session_results sr ON sp.session_id = sr.session_id AND sp.prize_id = sr.prize_id
    WHERE sp.session_id = ? AND sp.prize_id = ?
    GROUP BY sp.prize_id
  `).get(sessionId, prizeId);
  return result ? result.remaining : 0;
}

/**
 * 执行一次抽奖（含事务）
 */
export function executeDraw(sessionId, prizeId) {
  const db = getDb();

  const eligible = getEligibleParticipants(sessionId, prizeId);
  if (eligible.length === 0) {
    return { success: false, error: '没有可参与抽奖的人员' };
  }

  const remaining = getPrizeRemaining(sessionId, prizeId);
  if (remaining <= 0) {
    return { success: false, error: '该奖品已经抽完' };
  }

  const winner = weightedDraw(eligible);

  const drawResult = db.prepare(
    'INSERT INTO session_results (session_id, participant_id, prize_id) VALUES (?, ?, ?)'
  ).run(sessionId, winner.id, prizeId);

  db.prepare("UPDATE lottery_sessions SET status = '进行中' WHERE id = ?").run(sessionId);

  const prize = db.prepare('SELECT * FROM prizes WHERE id = ?').get(prizeId);

  return {
    success: true,
    result: {
      id: drawResult.lastInsertRowid,
      winner: { id: winner.id, name: winner.name, email: winner.email, phone: winner.phone, avatar: winner.avatar },
      prize: { id: prize.id, name: prize.name, description: prize.description, image: prize.image },
      drawn_at: new Date().toISOString()
    }
  };
}

/**

 */
export function resetSession(sessionId) {

  return { success: true, message: '场次已重置' };
}

export function getSessionStats(sessionId) {

    total_winners: totalWinners,

  };
```

## 10. 关键设计说明

### 10.1 技术选型理由

| 技术 | 用途 | 选型理由 |
|------|------|----------|
| Node.js + Express.js | Web 框架 | 轻量、生态丰富、适合实时应用 |
| better-sqlite3 | 数据库 | 同步 API 简单易用、无需独立数据库服务、嵌入式部署方便 |
| Socket.io | 实时通信 | 管理端与展示端实时同步、自动重连、房间机制 |
| multer | 文件上传 | Express 标准文件上传中间件 |
| xlsx | Excel 解析 | 支持批量导入参与者 |

### 10.2 数据库设计要点

- **WAL 模式**: 启用 Write-Ahead Logging，提升并发读写性能
- **外键约束**: `foreign_keys = ON`，确保数据完整性
- **触发器**: 自动同步参与者中奖状态，避免应用层遗漏
- **软删除 vs 硬删除**: 当前采用硬删除（CASCADE），通过外键自动清理关联数据
- **单行配置表**: `lottery_config` 使用 `id = 1` 固定行，简化配置读写

### 10.3 抽奖算法说明

采用 **加权随机算法**（Weighted Random Selection）：

1. 计算所有参与者权重总和 `totalWeight`
2. 生成 `[0, totalWeight)` 范围内的随机数
3. 遍历参与者，依次减去权重值，直到随机数 <= 0
4. 当前参与者即为中奖者

权重默认为 1（均等概率），可通过 API 调整个别参与者的中奖权重。

### 10.4 实时通信流程

```
管理端 (Admin)                    服务端 (Server)                   展示端 (Display)
    |                                  |                                  |
    |-- POST /api/console/start ----->|                                  |
    |                                  |-- emit draw:start ------------->|
    |                                  |                                  | (播放抽奖动画)
    |                                  |                                  |
    |-- POST /api/console/stop ------>|                                  |
    |                                  |-- 执行加权随机抽奖               |
    |                                  |-- emit draw:result ------------>|
    |<-- 返回中奖结果 ------------------|                                  | (展示中奖结果)
```
